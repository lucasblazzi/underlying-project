@startuml
!include https://raw.githubusercontent.com/kirchsth/C4-PlantUML/extended/C4_Container.puml
title Visão Geral
Person(User, "Usuário")
System_Boundary(c1, "Cloud", $link="https://github.com/plantuml-stdlib/C4-PlantUML") {
    rectangle Frontend{
    Container(web_app, "Aplicação Web","Underlying")
  }

    rectangle middle{
    Container(api, "API")
    Container(auth,"Autenticador")
  }

    rectangle Backend{
    Container(lambdas,"Serverless Functions")
    Container(user_lambdas,"User Functions","Dados referentes aos Clientes")
    ContainerDb(banco,"Banco de dados")
  }
}

System_Ext(b3,"Dados da B3")

Rel(User, Frontend, "Acessa")
BiRel_R(middle, Frontend," ")
BiRel(auth, api," ")
BiRel(middle, Backend," ")
BiRel(user_lambdas, banco," ")
Rel_R(Backend, b3, "Recebe dados")

@enduml
